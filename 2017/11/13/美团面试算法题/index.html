BinaryTreeNode*CreateCore(int*array,int begin,int end)

{  
    if(begin > end)  
        return NULL;  

    int mid = (begin+end)/2;  
    BinaryTreeNode* pRoot = CreateBinaryTreeNode(array[mid]);  
    pRoot->m_pLeft = CreateCore(array, begin, mid-1);  
    pRoot->m_pRight = CreateCore(array, mid+1, end);  

    return pRoot;  
}
